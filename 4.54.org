* Answer
#+begin_src scheme
  (define (install-require-package)
    (put 'eval 'require
         (lambda (exp)
           (let ((pproc (analyze (require-predicate exp))))
             (lambda (env succeed fail)
               (pproc env
                      (lambda (pred-value fail2)
                        (if (not (true? pred-value))
                            (fail2)
                            (succeed 'ok fail2)))
                      fail)))))
    'ok)

  (define (require-predicate exp)
    (cadr exp))
#+end_src
